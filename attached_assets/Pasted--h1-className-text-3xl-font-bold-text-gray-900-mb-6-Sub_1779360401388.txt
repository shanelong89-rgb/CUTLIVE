<h1 className="text-3xl font-bold text-gray-900 mb-6">Submissions</h1>

          {/* Tabs */}
          <div className="flex gap-2 mb-8">
            <TabButton active={activeTab === 'pending'} onClick={() => setActiveTab('pending')} count={pendingSubs.length} label="Pending" />
            <TabButton active={activeTab === 'approved'} onClick={() => setActiveTab('approved')} count={approvedSubs.length} label="Approved" />
            <TabButton active={activeTab === 'rejected'} onClick={() => setActiveTab('rejected')} count={rejectedSubs.length} label="Rejected" />
          </div>

          {/* Content */}
          {loading ? (
            <p className="text-gray-500 text-center py-20">Loading...</p>
          ) : currentSubs.length === 0 ? (
            <div className="text-center py-20">
              <p className="text-gray-500 text-lg">No {activeTab} submissions.</p>
            </div>
          ) : (
            <div className="space-y-4">
              {currentSubs.map(sub => (
                <div key={sub.id} className="bg-white rounded-lg border border-gray-200 p-6">
                  <div className="flex gap-6">
                    {sub.image && (
                      <img src={sub.image} alt="" className="w-32 h-32 object-cover rounded-lg bg-gray-100" />
                    )}
                    <div className="flex-1">
                      <div className="flex items-center gap-2 mb-2">
                        <span className={`text-xs px-2 py-1 rounded ${
                          sub.submission_type === 'instagram' ? 'bg-purple-100 text-purple-700' : 'bg-blue-100 text-blue-700'
                        }`}>
                          {sub.submission_type}
                        </span>
                        {sub.status === 'pending_scrape' && (
                          <span className="text-xs px-2 py-1 rounded bg-yellow-100 text-yellow-700">needs scrape</span>
                        )}
                      </div>
                      <h3 className="text-lg font-semibold text-gray-900">{sub.title || 'Untitled Event'}</h3>
                      <div className="text-sm text-gray-600 mt-2 space-y-1">
                        {sub.date && <p>📅 {sub.date} {sub.time && `@ ${sub.time}`}</p>}
                        {sub.venue && <p>📍 {sub.venue}</p>}
                        {sub.price && <p>💰 {sub.price}</p>}
                      </div>
                      {sub.instagram_url && (
                        <a href={sub.instagram_url} target="_blank" rel="noopener noreferrer" className="inline-flex items-center gap-1 text-sm text-purple-600 mt-3">
                          <ExternalLink className="w-4 h-4" />
                          View Instagram Post
                        </a>
                      )}
                    </div>
                    {activeTab === 'pending' && (
                      <div className="flex gap-2">
                        <button onClick={() => handleReject(sub.id)} className="flex items-center gap-1 px-4 py-2 border border-gray-300 rounded-lg text-sm hover:bg-gray-50">
                          <X className="w-4 h-4" /> Reject
                        </button>
                        <button onClick={() => handleApprove(sub)} className="flex items-center gap-1 px-4 py-2 bg-gray-900 text-white rounded-lg text-sm hover:bg-gray-800">
                          <Check className="w-4 h-4" /> Approve
                        </button>
                      </div>
                    )}
                  </div>
                </div>
              ))}
            </div>
          )}
        </div>
      </main>
    </div>
  );
}

function TabButton({ active, onClick, count, label }: { active: boolean; onClick: () => void; count: number; label: string }) {
  return (
    <button onClick={onClick} className={`px-5 py-2.5 rounded-lg font-medium transition-colors ${
      active ? 'bg-gray-900 text-white' : 'bg-white text-gray-700 border border-gray-200 hover:bg-gray-50'
    }`}>
      {label} ({count})
    </button>
  );
}
