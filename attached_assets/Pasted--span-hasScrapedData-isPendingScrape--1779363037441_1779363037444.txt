</span>
            )}
            
            {hasScrapedData && !isPendingScrape && (
              <span className="text-xs px-2 py-1 rounded bg-green-100 text-green-700">
                scraped
              </span>
            )}
          </div>

          {/* Title */}
          <h3 className="text-lg font-semibold text-gray-900">
            {displayTitle}
          </h3>

          {/* Details */}
          <div className="text-sm text-gray-600 mt-2 space-y-1">
            {displayDate && (
              <p>📅 {displayDate} {displayTime && `@ ${displayTime}`}</p>
            )}
            {displayVenue && <p>📍 {displayVenue}</p>}
            {displayPrice && <p>💰 {displayPrice}</p>}
          </div>

          {/* Scraped Data Preview (if available) */}
          {hasScrapedData && (
            <div className="mt-3 p-3 bg-gray-50 rounded-lg text-xs">
              <p className="font-medium text-gray-700 mb-1">Scraped Data:</p>
              <pre className="text-gray-600 overflow-x-auto whitespace-pre-wrap">
                {JSON.stringify(sub.scraped_data, null, 2)}
              </pre>
            </div>
          )}

          {/* IG Link */}
          {sub.instagram_url && (
            <a 
              href={sub.instagram_url}
              target="_blank"
              rel="noopener noreferrer"
              className="inline-flex items-center gap-1 text-sm text-purple-600 mt-3 hover:text-purple-800"
            >
              <ExternalLink className="w-4 h-4" />
              View Instagram Post
            </a>
          )}
        </div>

        {/* Actions */}
        {showActions && (
          <div className="flex gap-2 flex-shrink-0">
            {!isPendingScrape && (
              <>
                <button
                  onClick={() => onReject(sub.id)}
                  className="flex items-center gap-1 px-4 py-2 border border-gray-300 rounded-lg text-sm hover:bg-gray-50"
                >
                  <X className="w-4 h-4" />
                  Reject
                </button>
                <button
                  onClick={() => onApprove(sub)}
                  className="flex items-center gap-1 px-4 py-2 bg-gray-900 text-white rounded-lg text-sm hover:bg-gray-800"
                >
                  <Check className="w-4 h-4" />
                  Approve
                </button>
              </>
            )}
            {isPendingScrape && (
              <span className="text-sm text-gray-500 italic">
                Waiting for scrape...
              </span>
            )}
          </div>
        )}
      </div>
    </div>
  );
}

function TabButton({ 
  active, 
  onClick, 
  count, 
  label 
}: { 
  active: boolean; 
  onClick: () => void; 
  count: number; 
  label: string;
}) {
  return (
    <button
      onClick={onClick}
      className={`px-5 py-2.5 rounded-lg font-medium transition-colors ${
        active
          ? 'bg-gray-900 text-white'
          : 'bg-white text-gray-700 border border-gray-200 hover:bg-gray-50'
      }`}
    >
      {label} ({count})
    </button>
  );
}
